        	    method: 'POST',
        	    url: CONSTANTS.SERVICE_URL_BASE + '/auction/create-auction',
        	    data: data,
        	    headers: {
        	        "Content-Type": "application/json",
        	        'X-Access-Token': $cookies.get('opa.token'),
